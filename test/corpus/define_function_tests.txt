    VAR result: Int;
    IF (val < lo) {
        result = lo;
    } ELSE {
        IF (val > hi) {
            result = hi;
        } ELSE {

    VAR acc: Int = 1;
    VAR i: Int = 1;
    WHILE (i <= n) {
        acc = acc * i;
        i = i + 1;
