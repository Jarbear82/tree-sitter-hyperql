    STATE [
        total: Int = 0
    ];
    ACCUMULATE [
        total = total + val
    ];
    FINALIZE total;
};

DEFINE AGGREGATE FUNCTION @pure WeightedAvg(val: Float, weight: Float): Float {
    STATE [
        total_weighted: Float = 0.0,
        total_weight: Float = 0.0
    ];
    ACCUMULATE [
        total_weighted = total_weighted + val * weight,
        total_weight = total_weight + weight
    ];
    FINALIZE IF total_weight == 0.0 THEN 0.0 ELSE total_weighted / total_weight;
};

DEFINE AGGREGATE FUNCTION @pure CountPresent(val: String?): Int {
    STATE [
        count: Int = 0
    ];
    ACCUMULATE [
        count = IF val IS NULL THEN count ELSE count + 1
    ];
    FINALIZE count;
};
